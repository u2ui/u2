## loading.js

## Usage

```html
<button id="btn1" onclick="setTimeout(loading.mark(this),2000)">load</button>
```

## Install

```js
import * as module from "https://cdn.jsdelivr.net/gh/u2ui/u2@x.x.x/js/loading/loading.min.js"
```
